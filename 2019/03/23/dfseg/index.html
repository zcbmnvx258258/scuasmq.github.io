#算法鸭

#include <iostream>

using namespace std;

int num[100000];
int k=0;
int mov[4][2]={{1,0},{0,1},{-1,0},{0,-1}};
int a[10];
int map[10][10];

void dfs(int x,int y,int z);//x.y为横纵坐标,z为层数

int main(){
    for (int i=0; i<5; i++) {
        for (int j=0; j<5; j++) {
            cin>>map[i][j];
        }
        }
            for (int i=0; i<5 ; i++) {
            for (int j=0; j<5; j++) {
            a[0] = map[i][j];
            dfs(i, j, 1);
            }
            }
        cout<<k<<endl;
        return 0;
    }

void dfs(int x,int y,int z){
    if(z==6){
        int testnum = a[0]*100000+a[1]*10000+a[2]*1000+a[3]*100+a[4]*10+a[5];
        if(k==0){
            num[k++]=testnum;
            }   
        else{
            int flag=1;
            for (int i=0; i<k; i++) {
            if(num[i] == testnum){
            flag = 0;
            break;
                }
            }
        if(flag)
            num[k++] = testnum;
        }

   return;
    }
    for (int i=0; i<4; i++) {
        int cx = x+mov[i][0];
        int cy = y+mov[i][1];
        if(cx>=0&&cx<5&&cy>=0&&cy<5){
            a[z]=map[cx][cy];
            dfs(cx, cy, z+1);
        }
    }
}
